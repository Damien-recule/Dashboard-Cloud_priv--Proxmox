<script>

function updateStatus() {

  fetch("/status")

    .then(r => r.json())

    .then(data => {

      document.getElementById("dot-terraform").className =

        "dot " + (data.terraform ? "green" : "red");

      document.getElementById("dot-ansible").className =

        "dot " + (data.ansible ? "green" : "red");

      document.getElementById("dot-jenkins").className =

        "dot " + (data.jenkins ? "green" : "red");

      document.getElementById("dot-graylog").className =

        "dot " + (data.graylog ? "green" : "red");

    });

}

 

function loadVMs() {

  const node = document.getElementById("node").value;

 

  fetch("/vms/" + node)

    .then(r => r.json())

    .then(vms => {

      let html = "<table>";

      vms.forEach(vm => {

        html += `<tr>

          <td>${vm.vmid}</td>

          <td>${vm.name}</td>

          <td>${vm.status}</td>

        </tr>`;

      });

      html += "</table>";

 

      document.getElementById("vm-list").innerHTML = html;

    });

}

 

setInterval(updateStatus, 5000);

loadVMs();

</script>
